Завдання 1.9 - Підрахунок тегів

Умова: Поверніть об'єкт, в якому вказано кількість тегів. Очікуваний результат: {js: 3, nodejs: 3, html: 2, css: 2, react: 2}

Код програми:

const tweets = [ { id: "000", likes: 5, tags: ["js", "nodejs"] }, { id: "001", likes: 2, tags: ["html", "css"] }, { id: "002", likes: 17, tags: ["html", "js", "nodejs"] }, { id: "003", likes: 8, tags: ["css", "react"] }, { id: "004", likes: 0, tags: ["js", "nodejs", "react"] }, ]; function countTags() { console.log("=== ПІДРАХУНОК ТЕГІВ ==="); console.log("Масив твітів:"); console.log(tweets); console.log(""); // Об'єкт для зберігання кількості тегів const tagCount = {}; // Перебираємо всі твіти tweets.forEach(tweet => { console.log(`Твіт ID: ${tweet.id}, теги: ${tweet.tags.join(", ")}`); // Перебираємо всі теги в поточному твіті tweet.tags.forEach(tag => { // Якщо тег вже є - збільшуємо лічильник, інакше - встановлюємо 1 if (tagCount[tag]) { tagCount[tag]++; } else { tagCount[tag] = 1; } }); }); console.log(""); console.log("Результат підрахунку тегів:"); console.log(tagCount); console.log(""); console.log("Деталі:"); for (let tag in tagCount) { console.log(`${tag}: ${tagCount[tag]}`); } alert(`Знайдено унікальних тегів: ${Object.keys(tagCount).length}\n\nДив. Console для деталей`); return tagCount; }

Пояснення:

  • tagCount = {} - порожній об'єкт для підрахунку
  • forEach() - перебір масиву твітів та їх тегів
  • tagCount[tag] - доступ до властивості об'єкта
  • if (tagCount[tag]) - перевірка чи тег вже існує
  • tagCount[tag]++ - збільшення лічильника
  • for...in - перебір властивостей об'єкта
