Завдання 1.2 - getProductDetails з callback

Умова: Написати функцію getProductDetails, яка приймає ідентифікатор товару productId та дві колбек-функції successCallback та errorCallback. Функція повинна отримати деталі про вказаний товар та передати їх у successCallback. У випадку якщо товар не знайдено, викликається errorCallback.

Код програми:

// База даних товарів const products = [ { id: 101, name: "Ноутбук", price: 25000, category: "Електроніка" }, { id: 102, name: "Смартфон", price: 15000, category: "Електроніка" }, { id: 103, name: "Навушники", price: 2500, category: "Аксесуари" }, { id: 104, name: "Клавіатура", price: 1500, category: "Аксесуари" } ]; // Функція getProductDetails function getProductDetails(productId, successCallback, errorCallback) { console.log(`Шукаємо товар з ID: ${productId}`); // Пошук товару в базі даних const product = products.find(item => item.id === productId); // Якщо товар знайдено if (product) { successCallback(product); } else { // Якщо товар не знайдено errorCallback(`Товар з ID ${productId} не знайдено!`); } } // Callback для успішного результату function successCallback(product) { console.log("SUCCESS: Товар знайдено:"); console.log(`ID: ${product.id}`); console.log(`Назва: ${product.name}`); console.log(`Ціна: ${product.price} грн`); console.log(`Категорія: ${product.category}`); alert(`Товар знайдено!\n\nНазва: ${product.name}\nЦіна: ${product.price} грн`); } // Callback для помилки function errorCallback(errorMessage) { console.log(`ERROR: ${errorMessage}`); alert(errorMessage); } // Тестова функція function testProduct(productId) { console.log("=".repeat(50)); getProductDetails(productId, successCallback, errorCallback); console.log("=".repeat(50)); }
