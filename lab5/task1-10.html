Завдання 1.10 - Перевірка дужок

Умова: Напишіть функцію checkBrackets(str) яка приймає рядок js коду і перевіряє правильність закриття дужок () {} []. Якщо рядок містить коректний код функція повертає true. В іншому випадку повертає false.

Код програми:

function checkBrackets(str) { const stack = []; const brackets = { '(': ')', '{': '}', '[': ']' }; for (let char of str) { // Якщо відкриваюча дужка - додаємо в стек if (brackets[char]) { stack.push(char); } // Якщо закриваюча дужка else if (char === ')' || char === '}' || char === ']') { // Якщо стек порожній - помилка if (stack.length === 0) { return false; } // Беремо останню відкриваючу дужку const lastBracket = stack.pop(); // Перевіряємо чи відповідає закриваюча дужка if (brackets[lastBracket] !== char) { return false; } } } // Якщо стек не порожній - є незакриті дужки return stack.length === 0; } function testBrackets() { console.log("=== ПЕРЕВІРКА ДУЖОК ==="); console.log(""); // Тестові випадки const tests = [ { code: "function test() { return [1, 2, 3]; }", expected: true }, { code: "const arr = [1, 2, (3 + 4)]", expected: true }, { code: "if (x > 0) { console.log('positive'); }", expected: true }, { code: "function test() { return [1, 2, 3; }", expected: false }, { code: "const arr = [1, 2, (3 + 4]", expected: false }, { code: "if (x > 0 { console.log('error'); }", expected: false }, { code: "{{[[()]]}", expected: false }, { code: "{{[[(())]]}}", expected: true } ]; let passed = 0; let failed = 0; tests.forEach((test, index) => { const result = checkBrackets(test.code); const status = result === test.expected ? "PASSED" : "FAILED"; console.log(`Тест ${index + 1}: ${status}`); console.log(`Код: ${test.code}`); console.log(`Результат: ${result}, Очікувалось: ${test.expected}`); console.log(""); if (result === test.expected) { passed++; } else { failed++; } }); console.log("=== РЕЗУЛЬТАТИ ==="); console.log(`Успішно: ${passed}/${tests.length}`); console.log(`Провалено: ${failed}/${tests.length}`); alert(`Тестування завершено!\n\nУспішно: ${passed}/${tests.length}\nПровалено: ${failed}/${tests.length}\n\nДив. Console для деталей`); }

Пояснення:

  • stack - стек для зберігання відкриваючих дужок
  • brackets - об'єкт відповідності відкриваючих та закриваючих дужок
  • stack.push() - додавання відкриваючої дужки в стек
  • stack.pop() - видалення останньої дужки зі стеку
  • Алгоритм: кожна закриваюча дужка має відповідати останній відкриваючій
  • return stack.length === 0 - всі дужки мають бути закриті
