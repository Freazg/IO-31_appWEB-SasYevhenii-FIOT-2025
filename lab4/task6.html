Завдання 6 - Функція makeArray

Умова: Напиши функцію makeArray, яка приймає три параметри: firstArray (масив), secondArray (масив) і maxLength (число). Функція створює новий масив, який містить всі елементи з firstArray, а потім всі елементи з secondArray. Якщо кількість елементів перевищує maxLength, повернути копію масиву з довжиною maxLength елементів.

Код програми:

function makeArray(firstArray, secondArray, maxLength) { // Створити новий масив, об'єднавши firstArray та secondArray let newArray = firstArray.concat(secondArray); console.log(`Перший масив: [${firstArray}]`); console.log(`Другий масив: [${secondArray}]`); console.log(`Об'єднаний масив: [${newArray}]`); console.log(`Максимальна довжина: ${maxLength}`); // Якщо кількість елементів перевищує maxLength if (newArray.length > maxLength) { // Повернути копію масиву з довжиною maxLength елементів let result = newArray.slice(0, maxLength); console.log(`Результат (обрізано): [${result}]`); return result; } else { // В іншому випадку повернути весь новий масив console.log(`Результат (повний): [${newArray}]`); return newArray; } } // Тестова функція function testMakeArray() { console.log("=== Тестування функції makeArray ==="); console.log(""); // Тест 1 console.log("--- Тест 1 ---"); let result1 = makeArray(["Mango", "Poly"], ["Ajax", "Chelsea"], 3); console.log(`Повернутий масив: [${result1}]`); console.log(""); // Тест 2 console.log("--- Тест 2 ---"); let result2 = makeArray(["Mango", "Poly", "Houston"], ["Ajax", "Chelsea"], 4); console.log(`Повернутий масив: [${result2}]`); console.log(""); // Тест 3 console.log("--- Тест 3 ---"); let result3 = makeArray(["Mango"], ["Ajax", "Chelsea", "Poly", "Houston"], 3); console.log(`Повернутий масив: [${result3}]`); console.log(""); // Тест 4 console.log("--- Тест 4 ---"); let result4 = makeArray(["Earth", "Jupiter"], ["Neptune", "Mars", "Venus"], 2); console.log(`Повернутий масив: [${result4}]`); console.log(""); // Тест 5 console.log("--- Тест 5 ---"); let result5 = makeArray(["Earth", "Jupiter"], ["Neptune", "Mars", "Venus"], 8); console.log(`Повернутий масив: [${result5}]`); console.log(""); console.log("=== Тестування завершено ==="); alert("Тестування завершено! Дивіться результати в консолі."); }
