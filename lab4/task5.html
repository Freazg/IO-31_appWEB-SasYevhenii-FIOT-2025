Завдання 5 - Станція з продажу ремонтних дроїдів

Умова: Оголосити функцію makeTransaction(quantity, pricePerDroid, customerCredits), яка складає та повертає повідомлення про купівлю ремонтних дроїдів. Перевірити чи достатньо коштів у клієнта.

Код програми:

function makeTransaction(quantity, pricePerDroid, customerCredits) { // Оголосити змінну для зберігання загальної суми замовлення let totalPrice = quantity * pricePerDroid; console.log(`Кількість дроїдів: ${quantity}`); console.log(`Ціна за дроїда: ${pricePerDroid} credits`); console.log(`Загальна вартість: ${totalPrice} credits`); console.log(`Кредити клієнта: ${customerCredits} credits`); // Додати перевірку, чи зможе клієнт оплатити замовлення if (totalPrice > customerCredits) { // Якщо сума до сплати перевищує кількість кредитів console.log("Результат: Insufficient funds!"); return "Insufficient funds!"; } else { // В іншому випадку повернути повідомлення про успішну покупку let message = `You ordered ${quantity} droids worth ${totalPrice} credits!`; console.log(`Результат: ${message}`); return message; } } // Тестова функція function testTransaction() { console.log("=== Тестування функції makeTransaction ==="); console.log(""); // Тест 1 - Достатньо коштів console.log("--- Тест 1 ---"); let result1 = makeTransaction(5, 3000, 23000); alert(result1); console.log(""); // Тест 2 - Недостатньо коштів console.log("--- Тест 2 ---"); let result2 = makeTransaction(3, 1000, 15000); console.log(result2); console.log(""); // Тест 3 - Недостатньо коштів console.log("--- Тест 3 ---"); let result3 = makeTransaction(10, 5000, 8000); console.log(result3); console.log(""); // Тест 4 - Достатньо коштів console.log("--- Тест 4 ---"); let result4 = makeTransaction(8, 2000, 10000); console.log(result4); console.log(""); // Тест 5 - Рівно достатньо console.log("--- Тест 5 ---"); let result5 = makeTransaction(10, 500, 5000); console.log(result5); console.log(""); console.log("=== Тестування завершено ==="); }
