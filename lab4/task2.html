Завдання 2 - Визначення дня тижня

Умова: Змінна day може набувати одне із 7 значень: '1', '2'... '6' '7'. Якщо вона має значення '1', то у змінну finish запише понеділок, якщо має значення '2' – вівторок і так далі. Розв'язати завдання через switch-case.

Код програми:

function checkDay() { // Умова: Змінна day може набувати одне із 7 значень let day = prompt("Введіть номер дня тижня (1-7):"); // Перевірка на null або порожній рядок if (day === null || day === "") { alert("Ви не ввели номер дня!"); console.log("Користувач скасував введення"); return; } // Виведення у консоль console.log(`Введене значення: ${day}`); let finish; // Розв'язати завдання через switch-case switch(day) { case '1': finish = "Понеділок"; break; case '2': finish = "Вівторок"; break; case '3': finish = "Середа"; break; case '4': finish = "Четвер"; break; case '5': finish = "П'ятниця"; break; case '6': finish = "Субота"; break; case '7': finish = "Неділя"; break; default: finish = "Невірний номер дня! Введіть число від 1 до 7"; break; } // Значення змінної finish вивести в консоль console.log(`Результат: ${finish}`); alert(finish); }
