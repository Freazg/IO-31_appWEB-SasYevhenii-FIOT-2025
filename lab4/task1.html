Завдання 1 - Перевірка чверті години

Умова: Збережіть введене значення часу (год:хвил) у змінну hour. Вивести це значення у Console, використовуючи шаблонний рядок. Перевірити до якої чверті години відноситься введене значення.

Код програми:

function checkTime() { // Умова: Збережіть введене значення часу (год:хвил) let hour = prompt("Введіть час у форматі год:хвил (наприклад, 10:30):"); // Перевірка на null або порожній рядок if (hour === null || hour === "") { alert("Ви не ввели час!"); console.log("Користувач скасував введення або не ввів час"); return; } // Вивести це значення у Console, використовуючи шаблонний рядок console.log(`Введений час: ${hour}`); // Розділити час на години та хвилини let parts = hour.split(":"); // Перевірка правильності формату if (parts.length !== 2) { alert("Невірний формат часу! Використовуйте формат год:хвил"); console.log("Помилка формату"); return; } let hours = parseInt(parts[0]); let minutes = parseInt(parts[1]); // Перевірка коректності значень if (isNaN(hours) || isNaN(minutes) || hours < 0 || hours > 23 || minutes < 0 || minutes > 59) { alert("Невірні значення часу!"); console.log("Невірні значення годин або хвилин"); return; } // Перевірити до якої чверті години відноситься введене значення let quarter; if (minutes >= 0 && minutes < 15) { quarter = "Перша чверть години"; } else if (minutes >= 15 && minutes < 30) { quarter = "Друга чверть години"; } else if (minutes >= 30 && minutes < 45) { quarter = "Третя чверть години"; } else { quarter = "Четверта чверть години"; } // Відповідь вивести за допомогою alert() alert(quarter); console.log(`Результат: ${quarter}`); }
