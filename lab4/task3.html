Завдання 3 - Система авторизації

Умова: Є декілька користувачів User1, User2, User3, з відповідними паролями. Запитувати логін, перевіряти його та запитувати пароль. У випадку вдалої ідентифікації вивести привітання.

Код програми:

// Об'явити, проініціалізувати користувачів та паролі const users = { "User1": "pass1", "User2": "pass2", "User3": "pass3" }; function login() { // Запитати логін у користувача let login = prompt("Введіть логін (User1, User2 або User3):"); // Перевірка на Cancel або порожній рядок if (login === null || login === "") { alert("Cancelled"); console.log("Користувач скасував введення логіну"); return; } // Вивести логін у консоль використовуючи шаблонний рядок console.log(`Введений логін: ${login}`); // Перевірка чи існує такий користувач if (users.hasOwnProperty(login)) { // Запитати пароль let password = prompt(`Введіть пароль для ${login}:`); // Перевірка на Cancel if (password === null || password === "") { // Якщо нічого не ввели або натиснули Cancel // prompt знову просить ввести логін alert("Cancelled"); console.log("Користувач скасував введення пароля"); login(); // Рекурсивний виклик для повторного введення return; } console.log(`Введений пароль: ${"*".repeat(password.length)}`); // Перевірка пароля if (users[login] === password) { alert(`Hello, ${login}!`); console.log(`Успішна авторизація: Hello, ${login}!`); } else { alert("Невірний пароль!"); console.log("Невірний пароль"); } } else { // В іншому випадку вивести "I don't know you" alert("I don't know you"); console.log(`Невідомий користувач: ${login}`); } }
